%% Geometry of circles %%
% Question 2
\question Identify the centre and radius of the following circles:
\begin{parts}
	\part 
		\(
			x^{2} + y^{2} -2x -4y -20 =0
		\)
	\begin{solution}
		BE note: 
		\(
			(1 + 4 - r^{2} = -20)
		\)
		\[
			(x-1)^{2} + (y-2)^{2} = 5^{2}
		\]
		\qSolMath{
			\text{centre} \; 
			(1, 2)
			,\; \text{radius} = 
			5
		}{}
	\end{solution}

	\part 
		\(
			x^{2} + y^{2} -4x +6y +4 =0
		\)
	\begin{solution}
		BE note: 
		\(
			(4 + 9 - r^{2} = 4)
		\)
		\[
			(x-2)^{2} + (y+3)^{2} = 3^{2}
		\]
		\qSolMath{
			\text{centre} \; 
			(2, -3)
			,\; \text{radius} = 
			3
		}{}
	\end{solution}

	\part 
		\(
			x^{2} + y^{2} +2x -3 =0
		\)
	\begin{solution}
		BE note: 
		\(
			(1 - r^{2} = -3)
		\)
		\[
			(x+1)^{2} + (y+0)^{2} = 2^{2}
		\]
		\qSolMath{
			\text{centre} \; 
			(-1, 0)
			,\; \text{radius} = 
			2
		}{}
	\end{solution}

	\part 
		\(
			x^{2} + y^{2} +6x +7y - 14\frac{3}{4} =0
		\)
	\begin{solution}
		\par
		BE note: hate the use of mixed fractions in the question\dots
		\[
			x^{2} + y^{2} +6x +7y - \frac{59}{4} =0
		\]
		\[
			(x+3)^{2} + \left(y+\frac{7}{2}\right)^{2} = \; ?
		\]
		\[
			9 + \frac{49}{4} - r^{2} = -\frac{59}{4}
		\]
		\[
			r^{2} = \frac{85+59}{4}
			= \frac{144}{4}
			= 36
		\]
		\underline{Method 2:} equation is of the type
		\[
			x^{2} + y^{2} + 2gx + 2fy + c = 0
		\]
		Where
		\[
			g = 3, 
			\quad
			f = \frac{7}{2},
			\quad
			c = -\frac{59}{4}
		\]
		We know that
		\[
			c: (-g, -f), 
			\quad 
			r = \sqrt{g^{2} + f^{2} -c }
		\]
		\[
			c: \left( -3, -\frac{7}{2} \right), 
			\quad 
			r = \sqrt{
				3^{2} 
				+ \left( \frac{7}{2} \right)^{2} 
				- \left(-\frac{59}{4} \right) 
			}
		\]
		\[
			c: \left( -3, -\frac{7}{2} \right), 
			\quad 
			r 
			= \sqrt{
				\frac{144}{4}
			}
			= \sqrt{
				36
			}
		\]
		\qSolMath{
			\text{centre} \; 
			\left( -3, -\tfrac{7}{2} \right)
			,\; \text{radius} = 
			6
		}{}
	\end{solution}

	\part 
		\(
			3x^{2} + 3y^{2} -6x +9y +5 =0
		\)
	\begin{solution}
		\[
			x^{2} + y^{2} -2x +3y + \frac{5}{3} =0
		\]
		\[
			(x-1)^{2} + \left(y+\frac{3}{2}\right)^{2} = \; ?
		\]
		\[
			\rightarrow 
			1 + \frac{9}{4} - r^{2} = \frac{5}{3}
		\]
		\begin{align*}
			\rightarrow r^{2} 
				&= \frac{13}{4} - \frac{5}{3} 
				\\[0.2\baselineskip]
			r^{2}
				&= \frac{19}{12}
		\end{align*}
		\[
			r = \sqrt{\frac{19}{12}}
		\]
		\qSolMath{
			\text{centre} \; 
			\left( 1, -\tfrac{3}{2} \right)
			,\; \text{radius} = 
			\sqrt{\tfrac{19}{12}}
		}{}
	\end{solution}

\end{parts}
