% Question 12
\question What is the angle (in radians) subtended at the centre of a circle of 
radius 4cm by an arc of length 12cm?
\begin{solution}
	\par 
	We know that 1 rad corresponds to an arc of length $r$
	\[
		\therefore 
		\theta = \frac{12}{4}
	\]
	\qSolMath{
		= 3 \; \text{rad}
	}{}
\end{solution}

\appenddata{questionSolutions}{
{
$3 \; \text{rad}$
}
}
