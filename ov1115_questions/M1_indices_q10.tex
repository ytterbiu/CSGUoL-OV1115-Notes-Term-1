%% Powers or indices %%
% Question 10
\hlquestion Simplify each of the following expressions giving your answer in the 
form $Cx^{n}$, where $C$ and $n$ are numbers.
\begin{multicols}{3}
	\begin{parts}
		\part $3x^{2} \times 2x^{4}$
		\part $5x \times 4x^{5}$
		\part $(2x^{3})^{4}$
		
		\part $\tfrac{8x^{6}}{2x^{3}}$
		\part $\tfrac{3}{x^{2}} \times 4x^{5}$
		\part $12x^{8} \times \tfrac{1}{3x^{2}}$
		
		\part $(5x^{3})^{-1}$
		\part $(9x^{4})^{1/2}$
		\part $2x^{6} \times \tfrac{1}{4x^{-2}}$
		
		\part $2x^{4} \times \tfrac{1}{x^{5}}$
		\part $(2x)^{4} \times \tfrac{1}{x^{5}}$
		\part $6x^{3} \times \tfrac{1}{(2x)^{-1}}$
	\end{parts}
\end{multicols}

\begin{solution}
	\begin{multicols}{3}
		\begin{parts}
			% a
			\part $3x^{2} \times 2x^{4}$
			\[
				= (3 \times 2) \times (x^{2} \times x^{4})
			\]
			\qSolMath{
				6x^{6}
			}{}
			% b
			\part $5x \times 4x^{5}$
			\[
				= (5\times4)\times(x^{1} \times x^{5})
			\]
			\qSolMath{
				20x^{6}
			}{}
			% c
			\part $(2x^{3})^{4}$
			\[
				= 2^{4} x^{3\times4}
			\]
			\qSolMath{
				16 x^{12}
			}{}
			% d
			\part $\tfrac{8x^{6}}{2x^{3}}$
			\[
				= 4x^{6-3}
			\]
			\qSolMath{
				4x^{3}
			}{}
			% e
			\part $\tfrac{3}{x^{2}} \times 4x^{5}$
			\[
				= 3x^{-2} \times 4x^{5}
			\]
			\qSolMath{
				12x^{3}
			}{}
			% f
			\part $12x^{8} \times \tfrac{1}{3x^{2}}$
			\[
				= 12x^{8} \times \tfrac{1}{3} x^{-2}
			\]
			\qSolMath{
				4x^{6}
			}{}
			% g
			\part $(5x^{3})^{-1}$
			\[
				= \tfrac{1}{5x^{3}}
			\]
			\qSolMath{
				\tfrac{1}{5} x^{-3}
			}{}
			% h
			\part $(9x^{4})^{1/2}$
			\[
				= 9^{1/2} \times x^{\left(4 \times \tfrac{1}{2}\right)}
			\]
			\qSolMath{
				3x^{2}
			}{}
			% i
			\part $2x^{6} \times \tfrac{1}{4x^{-2}}$
			\[
				= 2x^{6} \times \tfrac{1}{4}x^{2}
			\]
			\qSolMath{
				\tfrac{1}{2} x^{8}
			}{}
			% j
			\part $2x^{4} \times \tfrac{1}{x^{5}}$
			\[
				= 2x^{4} \times x^{-5}
			\]
			\qSolMath{
				2x^{-1}
			}{}
			% k
			\part $(2x)^{4} \times \tfrac{1}{x^{5}}$
			\[
				= 16x^{4} \times x^{-5}
			\]
			\qSolMath{
				16x^{-1}
			}{}
			% l
			\part $6x^{3} \times \tfrac{1}{(2x)^{-1}}$
			\[
				= 6x^{3} \times 2x
			\]
			\qSolMath{
				12x^{4}
			}{}
		\end{parts}
	\end{multicols}
\end{solution}
