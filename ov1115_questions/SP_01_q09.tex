% Question 9
\question Calculate, to 1 decimal place, 
the length of the side marked $a$ in this triangle.
\par
\begin{center}
	\includegraphics[
		height = 35mm,width=1\textwidth,keepaspectratio
	]{%
		OV1115_Specimen_paper_1_Q9
	}
\end{center}
\begin{solution}
	SOH CAH TOA
	\[
		\cos{52^{\circ}} = \frac{8}{a}
	\]
	\[
		a = \frac{8}{\cos{52^{\circ}}}
	\]
	\[
		a = 12.994153\ldots \; \text{cm}
	\]
	\qSolMath{
		a = 13.0 \; \text{cm}
	}{
		(1dp)
	}
\end{solution}
