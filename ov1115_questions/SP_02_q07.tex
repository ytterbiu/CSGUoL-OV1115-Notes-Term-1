% Question 7
\question Power, $W$, is related to current, $I$, and resistance, $R$, 
in an electronic circuit by the formula: 
\[
	W = I^{2} R
\]
\begin{parts}
	\part Calculate $W$ when $I=5$ and $R=11.5$
		(Your answer should be to 1 decimal place but does not require units)
	\part Rearrange the formula to make $I$ the subject.
\end{parts}
\begin{solution}
	\begin{parts}
		\part %part a
			\[
				W = I^{2}R
			\]
			\[
				\rightarrow
				W = (5^{2}) (11.5)
			\]
			% \[
			% 	W = 287.5
			% \]
			\qSolMath{
				W = 287.5
			}{
				(1dp)
			}
		\part %part b
			\noindent
			\begin{align*}
				W 
					&= I^{2} R
					\\
				I^{2}
					&= \frac{W}{R}
				% 	\\
				% I
				% 	&= \sqrt{\frac{W}{R}}
			\end{align*}
			\qSolMath{
				I = \pm \sqrt{\frac{W}{R}}
			}{}
	\end{parts}
\end{solution}
