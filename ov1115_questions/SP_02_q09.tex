%% Specimen Paper 2 %%
% Question 9
\question Calculate, to 1 decimal place, 
the length of the side marked $a$ in this triangle.
\par
\begin{center}
	\includegraphics[
		height = 25mm,width=1\textwidth,keepaspectratio
	]{%
		OV1115_Specimen_paper_2_Q9
	}
\end{center}
\begin{solution}
	\textbf{SOH} CAH TOA
	\[
		\sin{(25^{\circ})} = \frac{15}{a}
	\]
	\[
		a = \frac{15}{\sin{(25^{\circ})}}
	\]
	\[
		a = 35.493023\ldots \; \text{cm}
	\]
	\qSolMath{
		a = 35.5 \; \text{cm}
	}{
		(1dp)
	}
\end{solution}
