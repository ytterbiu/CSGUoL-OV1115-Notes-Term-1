\question Find the areas of each of the triangles (to 3 decimal places) in 
\ref{subsec:triangle_formulae}, Question \ref{q:triangle_formulae_q01}.

BE note: from \ref{subsec:triangle_formulae} Q\ref{q:triangle_formulae_q01}:
\begin{multicols}{3}
	\begin{parts}
		\part $a=1,\, 	b=2,\, 		C=30^{\circ}$
		\part $a=3,\, 	c=4,\, 		B=50^{\circ}$
		\part $b=5,\, 	c=10,\, 	A=30^{\circ}$
	\end{parts}
\end{multicols}
\begin{solution}
	\[
		\text{area} \; [\alpha] = \frac{1}{2} a \cdot b \sin{C}
	\]
	\begin{multicols}{3}
		\setlength{\columnseprule}{0.4pt}
	\begin{parts}
		\part $a=1,\, 	b=2,\, 		C=30^{\circ}$
		\[
			\alpha = \frac{1}{2} (1)(2) \sin{30^{\circ}}
		\]
		\[
			\alpha = 0.5
		\]
		\qSolMath{
			0.500 \; \text{units}^{2}
		}{
			(3dp)
		}
		
		\part $a=3,\, 	c=4,\, 		B=50^{\circ}$
		\[
			\alpha = \frac{1}{2} (3)(4) \sin{50^{\circ}}
		\]
		\[
			\alpha = 4.596266\ldots
		\]
		\qSolMath{
			4.596 \; \text{units}^{2}
		}{
			(3dp)
		}
		
		\part $b=5$, $c=10$, $A=30^{\circ}$
		\[
			\alpha = \frac{1}{2} (5)(10) \sin{30^{\circ}}
		\]
		\[
			\alpha = 12.5
		\]
		\qSolMath{
			12.500 \; \text{units}^{2}
		}{
			(3dp)
		}
	\end{parts}
	\end{multicols}
