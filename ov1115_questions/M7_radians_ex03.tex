%% Radians %%
% Exercise 3
\question A sector of a circle is an area bounded by two radii and an arc. 
A sector has an angle at the centre of the circle. 
All the questions below relate to a circle with radius 5cm.
\begin{parts}
	\part Determine the length of the arc (correct to 2 decimal places) when 
	the angle at the centre is \label{W7RadiansEx3Q01}
	\begin{subparts}
		\subpart $1.2$ radians
		\subpart $\frac{\pi}{2}$ radians 
		\subpart $45^{\circ}$
	\end{subparts}
	\begin{solution}
		\[
			r = 5, 
			\quad
			s = \;?
		\]
		\begin{subparts}
			\subpart $\theta = 1.2 \; \text{rad}$
			\[
				s = (5)(1.2) \; \text{cm}
			\]
			\qSolMath{
				6 \; \text{cm}
			}{}

			\subpart $\theta = \frac{\pi}{2} \; \text{rad}$
			\[
				s = (5)(\tfrac{\pi}{2}) \; \text{cm}
			\]
			\[
				s = \frac{5\pi}{2} \; \text{cm}
			\]
			\[
				= 7.853981\dots
			\]
			\qSolMath{
				7.85 \; \text{cm}
			}{
				(2dp)
			}

			\subpart $\theta = 45^{\circ} = \frac{\pi}{4} \; \text{rad}$ 
			\[
				s = \frac{5\pi}{4} \; \text{cm}
			\]
			\[
				= 3.926990\dots
			\]
			\qSolMath{
				3.93 \; \text{cm}
			}{
				(2dp)
			}

		\end{subparts}
	\end{solution}

	\part Calculate the area (correct to 2 decimal places) of each of the three 
	sectors in (\ref{W7RadiansEx3Q01}).
	\begin{solution}
		\[
			r = 5, 
			\quad
			A = \frac{1}{2} r^{2} \theta
		\]
		Or can multiply all answers to \ref{W7RadiansEx3Q01} by $5/2$\dots
		\begin{subparts}
			\subpart $\theta = 1.2 \; \text{rad}$
			\[
				s = \left( \frac{25}{2} \right)(1.2) \; \text{cm}^{2}
			\]
			\qSolMath{
				15 \; \text{cm}^{2}
			}{}

			\subpart $\theta = \frac{\pi}{2} \; \text{rad}$
			\[
				s = 
				\left( \tfrac{25}{2} \right)(\tfrac{\pi}{2}) 
				\; \text{cm}^{2}
			\]
			\[
				s = \frac{25\pi}{4} \; \text{cm}^{2}
			\]
			\[
				= 19.63495\dots
			\]
			\qSolMath{
				19.63 \; \text{cm}^{2}
			}{
				(2dp)
			}

			\subpart $\theta = 45^{\circ} = \frac{\pi}{4} \; \text{rad}$ 
			\[
				s = 
				\left( \frac{25\pi}{8} \right)
				\; \text{cm}^{2}
			\]
			\[
				= 9.817477\dots
			\]
			\qSolMath{
				9.82 \; \text{cm}^{2}
			}{
				(2dp)
			}

		\end{subparts}
	\end{solution}

	\part A sector of this circle has area $50 \; \text{cm}^{2}$. 
	What is the angle (in radians) at the centre of this sector?
	\begin{solution}
		\[
			A = 50 \; \text{cm}^{2}
		\]
		We know that whole circle has an area of $25\pi \; \text{cm}^{2}$
		\[
			\frac{50}{25\pi} 
			= \frac{2}{\pi} 
			\; 
			\left( = 0.6366197\dots \right)
		\]
		This is $\equiv$ to angle of $2\pi$ radians
		\[
			\therefore
			(\tfrac{2}{\pi})(2\pi)
		\]
		\qSolMath{
			4 \; \text{rad}
		}{}
		OR using equation~\ref{eq:circleSectorArea}
		\[
			A = \frac{1}{2} r^{2} \theta
		\]
		\[
			A = 12.5 \theta
		\]
		If $A = 50 \; \text{cm}^{2}$
		\[
			\theta = \frac{50}{12.5} \; \text{rad}
		\]
		\qSolMath{
			\theta = 4 \; \text{rad}
		}{}
	\end{solution}

\end{parts}
