% Question 8
\question Transform the following formula to make $p$ the subject
\[
	\left( p + \frac{a}{V^{2}} \right) \left( V - b \right)
	=
	nRT
\]
\begin{solution}
	\[
		\left( p + \frac{a}{V^{2}} \right) \left( V - b \right)
		=
		nRT
	\]
	\[
		\rightarrow
		p + \frac{a}{V^{2}}
		=
		\frac{nRT}{\left( V - b \right)}
	\]
	\[
		\rightarrow
		p 
		=
		\frac{nRT}{\left( V - b \right)} - \frac{a}{V^{2}}
	\]
	\qSolMath{
		p
		=
		\frac{nRT}{\left( V - b \right)} - \frac{a}{V^{2}}
	}{}
	or
	\[
		p 
		= 
		- \frac{-ab + aV - nRTV^{2}}{V^{2} (V-b)}
	\]
\end{solution}
