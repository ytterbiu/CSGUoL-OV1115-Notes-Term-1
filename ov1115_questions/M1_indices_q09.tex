%% Powers or indices %%
% Question 09
\question Each of the following expressions can be written as 
$a^{n}$ for some value of $n$. 
In each case determine the value of $n$.
\begin{multicols}{4}
	\begin{parts}
		\part $a \times a \times a \times a$
		\part $\tfrac{1}{a \times a \times a}$
		\part $1$
		
		\hlpart $\sqrt[3]{a^{5}}$
		\hlpart $a^{3} \times a^{5}$
		\hlpart $\tfrac{a^{6}}{a^{2}}$
		
		\hlpart $(a^{4})^{2}$
		\hlpart $\tfrac{a^{2} \times a^{5}}{(a^{3})^{3}}$
		\hlpart $\sqrt{a} \times \tfrac{1}{(a^{-2})^{3}}$
		
		\hlpart $a^{1/2} \times \tfrac{1}{a^{2}}$
		\hlpart $\tfrac{1}{a^{-3}} \times \tfrac{1}{a^{-2}}$
		\hlpart $\tfrac{1}{(a^{-2})^{3}}$
	\end{parts}
\end{multicols}

\begin{solution}
	\begin{multicols}{4}
		\begin{parts}
			\part $a \times a \times a \times a$
			\[
				= a^{4}
			\]
			\qSolMath{
				\therefore
				n = 4
			}{}

			\part $\tfrac{1}{a \times a \times a}$
			\[
				= a^{-3}
			\]
			\qSolMath{
				\therefore
				n = -3
			}{}

			\part $1$
			\[
				= a^{0}
			\]
			\qSolMath{
				\therefore
				n = 0
			}{}
			% d
			\part $\sqrt[3]{a^{5}}$
			\[
				= a^{\tfrac{5}{3}}
			\]
			\qSolMath{
				\therefore
				n = \tfrac{5}{3}
			}{}
			% e
			\part $a^{3} \times a^{5}$
			\[
				= a^{8}
			\]
			\qSolMath{
				\therefore
				n = 8
			}{}
			% f
			\part $\tfrac{a^{6}}{a^{2}}$
			\[
				= a^{4}
			\]
			\qSolMath{
				\therefore
				n = 4
			}{}
			% g
			\part $(a^{4})^{2}$
			\[
				= a^{8}
			\]
			\qSolMath{
				\therefore
				n = 8
			}{}
			% h
			\part $\tfrac{a^{2} \times a^{5}}{(a^{3})^{3}}$
			\[
				= \tfrac{a^{7}}{a^{9}} 
				= a^{-2}
			\]
			\qSolMath{
				\therefore
				n = -2
			}{}
			% i
			\part $\sqrt{a} \times \tfrac{1}{a^{-2}}$
			\[
				= a^{\tfrac{1}{2}} \times a^{2}
				= a^{\tfrac{5}{2}}
			\]
			\qSolMath{
				\therefore
				n = \tfrac{5}{2}
			}{}
			% j
			\part $a^{1/2} \times \tfrac{1}{a^{2}}$
			\[
				= a^{\tfrac{5}{2}}
			\]
			\qSolMath{
				\therefore
				n = \tfrac{5}{2}
			}{}
			% k
			\part $\tfrac{1}{a^{-3}} \times \tfrac{1}{a^{-2}}$
			\[
				= a^{3} \times a^{2}
				= a^{5}
			\]
			\qSolMath{
				\therefore
				n = 5
			}{}
			% l
			\part $\tfrac{1}{(a^{-2})^{3}}$
			\[
				= \tfrac{1}{a^{-6}}
				= a^{6}
			\]
			\qSolMath{
				\therefore
				n = 6
			}{}
		\end{parts}
	\end{multicols}
\end{solution}
