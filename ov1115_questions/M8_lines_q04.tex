%% Equations of straight lines %%
% Question 4
\question Find the equation of the lines described below. 
Give the equation in the form:
\[
	y = mx + c
\]
\begin{parts}
	\part passing through $(4, 6)$ and $(8, 26)$,
		\begin{solution}
			% \par
			% Solution is of the type
			\[
				\frac{y-y_{1}}{y_{2}-y_{1}} = \frac{x-x_{1}}{x_{2}-x{1}}
			\]
			\[
				\rightarrow
				\frac{y-6}{26-6} = \frac{x-4}{8-4}
			\]
			\[
				\frac{y-6}{20} = \frac{x-4}{4}
			\]
			\[
				y-6 = \frac{20}{4} \cdot (x-4)
			\]
			\[
				y = 5x-20 +6
			\]
			\qSolMath{
				y = 5x - 14
			}{}
		\end{solution}
		
	\part passing through $(1, 1)$ and $(4, -8)$,
		\begin{solution}
			% \par
			% Solution is of the type
			\[
				\frac{y-y_{1}}{y_{2}-y_{1}} = \frac{x-x_{1}}{x_{2}-x{1}}
			\]
			\[
				\rightarrow
				\frac{y-1}{-8-1} = \frac{x-1}{4-1}
			\]
			\[
				\frac{y-1}{-9} = \frac{x-1}{3}
			\]
			\[
				y-1 = \frac{-9}{3} \cdot (x-1)
			\]
			\[
				y = -3x + 3 + 1
			\]
			\qSolMath{
				y = -3x + 4
			}{}
		\end{solution}

	\part passing through $(3, 4)$ and $(5, 4)$,
		\begin{solution}
			% \par
			% Solution is of the type
			\[
				\frac{y-y_{1}}{y_{2}-y_{1}} = \frac{x-x_{1}}{x_{2}-x{1}}
			\]
			\[
				\rightarrow
				\frac{y-4}{4-4} = \frac{x-3}{5-3}
			\]
			\[
				\frac{y-4}{0} = \frac{x-3}{2}
			\]
			\[
				\rightarrow
				\text{undefined.}
			\]
			At all points $y=4$\dots
			\par
			\qSolMath{
				y = 4
			}{}
		\end{solution}

	\part passing through $(0, 2)$ and $(4, 0)$,
		\begin{solution}
			% \par
			% Solution is of the type
			\[
				\frac{y-y_{1}}{y_{2}-y_{1}} = \frac{x-x_{1}}{x_{2}-x{1}}
			\]
			\[
				\rightarrow
				\frac{y-2}{0-2} = \frac{x-0}{4-0}
			\]
			\[
				\frac{y-2}{0-2} = \frac{x}{4}
			\]
			\[
				y-2 = \frac{-2}{4} \cdot (x)
			\]
			\qSolMath{
				y = -\tfrac{1}{2}x + 2
			}{}
		\end{solution}

	\part passing through $(-2, 3)$ and $(2, -5)$,
		\begin{solution}
			% \par
			% Solution is of the type
			\[
				\frac{y-y_{1}}{y_{2}-y_{1}} = \frac{x-x_{1}}{x_{2}-x{1}}
			\]
			\[
				\rightarrow
				\frac{y-3}{-5-3} = \frac{x-(-2)}{2-(-2)}
			\]
			\[
				\frac{y-3}{-8} = \frac{x+2}{4}
			\]
			\[
				y-3 = \frac{-8}{4} \cdot (x+2)
			\]
			\[
				y = -2x -4 +3
			\]
			\qSolMath{
				y = -2x -1
			}{}
		\end{solution}

\end{parts}

	\part $y = -3x + 4$
	\part $y = 4$
	\part $y = -\tfrac{1}{2}x + 2$
	\part $y = -2x -1$
\end{parts}
}
