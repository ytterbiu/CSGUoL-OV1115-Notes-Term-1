\question Evaluate $125^{-\frac{4}{3}}$ leaving your answer as a proper fraction 
or a whole number
\begin{solution}
	\[
		125^{-\frac{4}{3}} = \frac{1}{125^{\tfrac{4}{3}}}
	\]
	\[
		= \frac{1}{ \left( \sqrt[3]{125} \right)^{4} }
	\]
	\[
		= \frac{1}{ \left( 5 \right)^{4} }
	\]
	\qSolMath{
		= \frac{1}{625}
	}{}
\end{solution}

{
$\frac{1}{625}$
}
