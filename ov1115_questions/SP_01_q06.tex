% Question 6
\question Solve the following pair of simultaneous equations by a method of 
your choice
\[
	3a + 4b = 13
\]
\[
	a + 2b = 7
\]
\begin{solution}
	\[
		3a + 4b = 13 
		\quad (1)
	\]
	\[
		a + 2b = 7 
		\quad (2)
	\]
	\[
		\rightarrow
		a = 7-2b
		\quad (3)
	\]
	Sub (3) into (1)
	\noindent
	\begin{align*}
		3a + 4b 
			&= 13 
			\quad (1)
			\\
		\rightarrow 
		3(7-2b) + 4b 
			&= 13 
			\\
		21 - 6b + 4b 
			&= 13 
			\\
		21 - 13 
			&= 2b
			\\
		8
			&= 2b
			\\
		b
			&= 4 
			\quad (4)
	\end{align*}
	Sub (4) into (2)
	\[
		a + 2(4) = 7
	\]
	\[
		a = -1
	\]
	Check in (1), LHS:
	\[
		3(-1) + 4(4)
	\]
	\[
		-3 + 16 = 13 = \text{RHS}
	\]
	\qSolMath{
		b = 4,
		\quad 
		a = -1
	}{}
\end{solution}

\appenddata{questionSolutions}{
{
$a = -1$, $b=4$
}
}
