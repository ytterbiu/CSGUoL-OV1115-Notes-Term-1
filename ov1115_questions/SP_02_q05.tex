\question Solve the following equation: 
$\dfrac{x+2}{3} = \dfrac{1-2x}{5}$
\begin{solution}
	\[
		\dfrac{x+2}{3} = \dfrac{1-2x}{5}
	\]
	\[
		5(x+2) = 3(1-2x)
	\]
	\[
		5x + 10 = 3 - 6x
	\]
	\[
		11x = -7
	\]
	\qSolMath{
		x = - \frac{7}{11}
	}{}
\end{solution}

{
$x = - \frac{7}{11}$
}
