% Question 2
\question Evaluate $\left( \frac{8}{5} \right)^{3}$ leaving your answer as a
proper fraction or a whole number
\begin{solution}
	\[
		\left( \frac{8}{5} \right)^{3} 
		=
		\frac{8^{3}}{5^{3}} 
	\]
	\qSolMath{
		= \frac{512}{125}
	}{}
	Or
	\qSolMath{
		= 	4+\frac{12}{125}
	}{}
\end{solution}

\appenddata{questionSolutions}{
{
$\tfrac{512}{125}$ or $4+\frac{12}{125}$
}
}
