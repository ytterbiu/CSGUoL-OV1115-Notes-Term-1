\newcommand{\MathsTwoLogsQFourWkg}[5]{
	\par
	\(
		{#1}^{#2} = #3
	\)
	\[
		\rightarrow
		\log_{#1}{\left(#3\right)} = #2
	\]
	\[
		=#4\dots
	\]
	\qSolMath{
		#5
	}{
		(3dp)
	}
}
\newcommand{\MathsTwoLogsQFourQs}[5]{
	\(
		{#1}^{#2} = #3
	\)
}
\newcommand{\MathsTwoLogsQFourSols}[6][(3dp)]{
	$#6$ #1
}

\question Each of the following expressions can be simplified to $\log N$.
Determine the value of N in each case. 
We have not explicitly written down the base. 
You can assume the base is 10, but the results are identical whichever 
base is used.
\begin{multicols}{4}
	\begin{parts}
		\part \MathsTwoLogsQFourQs{10}{x}{5}{0.69897}{0.699}
		\part \MathsTwoLogsQFourQs{e}{x}{8}{2.07944}{2.079}
		\part \MathsTwoLogsQFourQs{10}{x}{\frac{1}{2}}{-0.3010299}{-0.301}
		% d-f
		\part \MathsTwoLogsQFourQs{e}{x}{0.1}{-2.302585}{-2.303}
		\part \MathsTwoLogsQFourQs{4}{x}{12}{1.792481}{1.792}
		\part \MathsTwoLogsQFourQs{3}{x}{2}{0.63092975}{0.631}
		% g-i
		\part \MathsTwoLogsQFourQs{7}{x}{1}{0}{0}
		\part \MathsTwoLogsQFourQs{\left( \frac{1}{2} \right)}{x}
			{\frac{1}{100}}{6.643856}{6.644}
		\hlpart \MathsTwoLogsQFourQs{\pi}{x}{10}{2.01146586}{2.011}
		% j-l
		\hlpart \MathsTwoLogsQFourQs{e}{x}{\pi}{1.1447298}{1.145}
		\hlpart \MathsTwoLogsQFourQs{\left( \frac{1}{3} \right)}{x}
		{2}{-0.6309297}{-0.631}
		\hlpart \MathsTwoLogsQFourQs{e}{x}{\pi}{1.1447298}{1.145}
	\end{parts}
\end{multicols}
\newpage
\begin{solution}
	\begin{multicols}{2}
		\begin{parts}
			\part \MathsTwoLogsQFourWkg{10}{x}{5}{0.69897}{0.699}
			\part \MathsTwoLogsQFourWkg{e}{x}{8}{2.07944}{2.079}
			\part \MathsTwoLogsQFourWkg{10}{x}{\frac{1}{2}}{-0.3010299}{-0.301}
			% d-f
			\part \MathsTwoLogsQFourWkg{e}{x}{0.1}{-2.302585}{-2.303}
			\part \MathsTwoLogsQFourWkg{4}{x}{12}{1.792481}{1.792}
			\part \MathsTwoLogsQFourWkg{3}{x}{2}{0.63092975}{0.631}
			% g-i
			\part \MathsTwoLogsQFourWkg{7}{x}{1}{0}{0}
			\part \MathsTwoLogsQFourWkg{\left( \frac{1}{2} \right)}{x}
				{\frac{1}{100}}{6.643856}{6.644}
			\hlpart \MathsTwoLogsQFourWkg{\pi}{x}{10}{2.01146586}{2.011}
			% j-l
			\hlpart \MathsTwoLogsQFourWkg{e}{x}{\pi}{1.1447298}{1.145}
			\hlpart \MathsTwoLogsQFourWkg{\left( \frac{1}{3} \right)}{x}
			{2}{-0.6309297}{-0.631}
			% \hlpart \MathsTwoLogsQFourWkg{e}{x}{\pi}{1.1447298}{1.145}
			\hlpart $10^{x} = e^{2x-1}$
				% \[
				% 	\rightarrow
				% 	\ln{10^{x}} 
				% 	= 
				% 	\ln{ e^{2x-1} }
				% \]
				\[
					\rightarrow
					\ln{10^{x}} 
					= 
					2x-1
					\rightarrow
					x\ln{10} = 2x -1
				\]
				% \[
				% 	x\ln{10} = 2x -1
				% \]
				% \[
				% 	x(\ln{10} - 2) = -1
				% \]
				% \[
				% 	x^{-1} = 2 - \ln{10}
				% \]
				% \[
				% 	x = -(\ln{10} - 2)^{-1}
				% \]
				% \[
				% 	x\ln{10} = 2x -1
				% \]
				% \[
				% 	\ln{10} = 2 - x^{-1}
				% 	\;\; \text{or} \;\;
				% 	x\ln{10} - 2x = -1
				% \]
				% \[
				% 	x^{-1} = 2 - \ln{10}
				% 	\;\; \text{or} \;\;
				% 	x(\ln{10} - 2) = -1
				% \]
				% \[
				% 	x = (2 - \ln{10})^{-1}
				% 	\;\; \text{or} \;\;
				% 	-(\ln{10} - 2)^{-1}
				% \]
				% \[
				% 	x = -3.30485547\dots
				% \]
				\[
					x 
					= 
					-(\ln{10} - 2)^{-1}
					= -3.304855\dots
				\]
				\qSolMath{
					-3.305
				}{
					(3dp)
				}
		\end{parts}
	\end{multicols}
\end{solution}
