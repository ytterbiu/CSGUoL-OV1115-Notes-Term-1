% custom commands to make life easier
\newcommand{\MathsTwoLogsQTwoWkg}[3]{
	\par
	% \centering
	% #1^{#2} = #3
	% \rightarrow
	\(
		\log_{#1}{#2}
		\rightarrow
		#1^{n} = #2
	\)
	\par
	$\rightarrow \log_{#1}{#2}=$%
	\colorbox{yellow}{%
		$#3$%
	}
}
\newcommand{\MathsTwoLogsQTwoQs}[3]{
	\(
		\log_{#1}{#2}
	\)
}
\newcommand{\MathsTwoLogsQTwoSols}[3]{
	$#3$
}
