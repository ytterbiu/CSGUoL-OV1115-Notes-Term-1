\question Convert an angle of $\dfrac{\pi}{10}$ radians to degrees.
\begin{solution}
	\[
		1 \; \text{rad} = \frac{180}{\pi}^{\circ}
	\]
	\[
		\rightarrow
		\frac{\pi}{10} \; \text{rad} 
		=
		\left( \frac{\pi}{10} \right) \left( \frac{180}{\pi} \right)^{\circ}
	\]
	\qSolMath{
		= 18^{\circ}
	}{}
\end{solution}

\appenddata{questionSolutions}{
{
$18^{\circ}$
}
