%% Equations of straight lines %%
% Question 3
\question Find the equation of the lines described below. 
Give the equation in the form:
\[
	y = mx + c
\]
\begin{parts}
	\part gradient 3, passing through (1, 4);
		\begin{solution}
			% \[
			% 	y = mx + c
			% \]
			Sub in values to find $c$
			\[
				\rightarrow
				(4) = \left( 3 \right) (1) + c
			\]
			\[
				c = 1
			\]
			\qSolMath{
				\rightarrow
				y = 3x +1
			}{}
		\end{solution}

	\part gradient -2, passing through (2, 0);
		\begin{solution}
			% \[
			% 	y = mx + c
			% \]
			% Sub in values to find $c$
			\[
				\rightarrow
				(0) = \left( -2 \right) (2) + c
			\]
			\[
				c = 4
			\]
			\qSolMath{
				\rightarrow
				y = -2x +4
			}{}
		\end{solution}
	
	\part gradient \( \frac{2}{5} \), passing through (5, -1);
		\begin{solution}
			% \[
			% 	y = mx + c
			% \]
			% Sub in values to find $c$
			\[
				\rightarrow
				(-1) = \left( \tfrac{2}{5} \right) (5) + c
			\]
			% \[
			% 	-1 - 2 = c
			% \]
			\[
				c = -3
			\]
			\qSolMath{
				\rightarrow
				y = \tfrac{2}{5} x - 3
			}{}
		\end{solution}

	\part gradient 0, passing through (-1, 2);
		\begin{solution}
			% \[
			% 	y = mx + c
			% \]
			% Sub in values to find $c$
			\[
				\rightarrow
				(2) = \left( 0 \right) (-1) + c
			\]
			\[
				c = 2
			\]
			\qSolMath{
				\rightarrow
				y = 2
			}{}
		\end{solution}

	\part gradient -1, passing through (1, -1);
		\begin{solution}
			% \[
			% 	y = mx + c
			% \]
			% Sub in values to find $c$
			\[
				\rightarrow
				(-1) = \left( -1 \right) (1) + c
			\]
			\[
				c = 0
			\]
			\qSolMath{
				\rightarrow
				y = -x
			}{}
		\end{solution}
\end{parts}

\appenddata{questionSolutions}{
{
\begin{parts}
	\part $y = 3x +1$
	\part $y = -2x +4$
	\part $y = \tfrac{2}{5} x - 3$
	\part $y = 2$
	\part $y = -x$
\end{parts}
}
}
