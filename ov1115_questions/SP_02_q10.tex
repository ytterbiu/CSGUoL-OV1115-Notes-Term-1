\question Convert an angle of $\dfrac{\pi}{4}$ radians to degrees.
\begin{solution}
	\[
		1 \; \text{rad} = \frac{180}{\pi}^{\circ}
	\]
	\[
		\rightarrow
		\frac{\pi}{4} \; \text{rad} 
		=
		\left( \frac{\pi}{4} \right) \left( \frac{180}{\pi} \right)^{\circ}
	\]
	\qSolMath{
		= 45^{\circ}
	}{}
\end{solution}

\appenddata{questionSolutions}{
{
$45^{\circ}$
}
