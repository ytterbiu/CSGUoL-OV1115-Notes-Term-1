% made custom command to make life easier
\newcommand{\MathsTwoLogsQOneWkg}[3]{
	\par
	$%\displaystyle 
	#1^{#2} = #3
	\rightarrow$
	\colorbox{yellow}{%
		$\log_{#1}{#3}=#2$%
	}
}
\newcommand{\MathsTwoLogsQOneQs}[3]{
	$#1^{#2} = #3$
}
\newcommand{\MathsTwoLogsQOneSols}[3]{
	$\log_{#1}{#3}=#2$%
}
