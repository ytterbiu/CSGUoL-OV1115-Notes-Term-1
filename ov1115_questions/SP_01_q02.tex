\question Evaluate $\left( \frac{4}{3} \right)^{4}$ leaving your answer as a
proper fraction or a whole number
\begin{solution}
	\[
		\left( \frac{4}{3} \right)^{4} 
		=
		\frac{4^{4}}{3^{4}} 
	\]
	\qSolMath{
		= \frac{256}{81}
	}{}
	Or
	\qSolMath{
		= 	3+\frac{13}{81}
	}{}
\end{solution}

\appenddata{questionSolutions}{
{
$\dfrac{256}{81}$ or $3+\dfrac{13}{81}$
}
}
