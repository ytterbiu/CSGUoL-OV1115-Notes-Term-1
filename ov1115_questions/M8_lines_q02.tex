%% Equations of straight lines %%
% Question 2
\question Find the equation of the lines described below. 
Give the equation in the form:
\[
	y = mx + c
\]
% \begin{multicols}{1}
	\begin{parts}
		\part gradient 5, y-intercept 3; 
		\part gradient -2, y-intercept -1;
		\part gradient 3, passing through the origin;
		\part gradient $\frac{1}{3}$ passing through (0, 1);
		\part gradient $-\frac{3}{4}$ , y-intercept $\frac{1}{2}$	
	\end{parts}
% \end{multicols}
\begin{solution}
	\begin{multicols}{2}
		\begin{parts}
			\part g: 5, y-intercept 3;
				\qSol{
					$y = 5x + 3$
				}{}
			\part g: -2, y-intercept -1;
				\qSol{
					$y = -2x + -1$
				}{}
			\part g: 3, passing through the origin;
				\qSol{
					$y = 3x$
				}{}
			\columnbreak
			\part g: $\frac{1}{3}$ passing through (0, 1);
				\qSol{
					$y = \frac{1}{3}x + 1$
				}{}
			\part g: $-\frac{3}{4}$ , y-intercept $\frac{1}{2}$	
				\qSol{
					$y = -\frac{3}{4}x + \frac{1}{2}$
				}{}
		\end{parts}
	\end{multicols}
\end{solution}

\end{parts}
}
