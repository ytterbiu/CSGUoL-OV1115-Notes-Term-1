%% Geometry of circles %%
% Question 1
\question Find the equation of the circle with given centre and radius
\begin{parts}
	\part centre (3,5), radius 3;
	\begin{solution}
		% \par
		% Solution is of the type
		\[
			(x-a)^{2} + (y-b)^{2} = r^{2}
		\]
		\[
			(x-3)^{2} + (y-5)^{2} = 3^{2}
		\]
		\[
			(x^2 - 6x + 9) + (y^{2} - 10y + 25) = 9 
		\]
		\qSolMath{
			x^{2} + y^{2} - 6x - 10y + 25 = 0
		}{}
	\end{solution}

	\part centre (-2,3), radius 1;
	\begin{solution}
		\[
			(x+2)^{2} + (y-3)^{2} = 1^{2}
		\]
		\[
			(x^2 +4x + 4) + (y^{2} - 6y + 9) = 1 
		\]
		\qSolMath{
			x^{2} + y^{2} + 4x - 6y + 12 = 0
		}{}
	\end{solution}

	\part centre (-1,-3), radius 2;
	\begin{solution}
		\[
			(x+1)^{2} + (y-3)^{2} = 2^{2}
		\]
		\[
			(x^2 +2x + 1) + (y^{2} - 6y + 9) = 4 
		\]
		\qSolMath{
			x^{2} + y^{2} +2x -6y +6 = 0
		}{}
	\end{solution}

	\part centre (2,-2), radius 5;
	\begin{solution}
		\[
			(x-2)^{2} + (y+2)^{2} = 5^{2}
		\]
		\[
			(x^2 -4x +4) + (y^{2} +4y +4) = 25 
		\]
		\qSolMath{
			x^{2} + y^{2} -4x +4y -17 = 0
		}{}
	\end{solution}
	
	\part centre (0,5), radius 4.
	\begin{solution}
		\[
			(x)^{2} + (y-5)^{2} = 4^{2}
		\]
		\[
			(x^2) + (y^{2} -10y +25) = 16 
		\]
		\qSolMath{
			x^{2} + y^{2} -10y +9 = 0
		}{}
	\end{solution}
\end{parts}
