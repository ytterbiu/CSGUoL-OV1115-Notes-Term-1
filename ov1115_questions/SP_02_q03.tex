\question Determine the value of $\log_{c}\sqrt{c}$
\begin{solution}
	\[
		\log_{c}{\sqrt{c}}
		=
		\log_{c}{c^{\frac{1}{2}}}
	\]
	\[
		\log_{10}{1000} = 3
	\]
	\[
		\rightarrow
		\log_{c}{c^{\frac{1}{2}}}
		= 
		\frac{1}{2}
	\]
	\qSolMath{
		\frac{1}{2}
	}{}
\end{solution}
