% Question 8
\question Transform the following formula to make $u$ the subject
\[
	v^{2} = u^{2} + 2as
\]
\begin{solution}
	\[
		v^{2} = u^{2} + 2as
	\]
	\[
		\rightarrow
		u^{2} = v^{2} - 2as
	\]
	\qSolMath{
		u = \sqrt{v^{2} - 2as}
	}{}
\end{solution}

\appenddata{questionSolutions}{
{
$u = \sqrt{v^{2} - 2as}$
}
}
