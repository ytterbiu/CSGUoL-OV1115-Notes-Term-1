%% Triangle Formulae %%
% Exercise 2 - Q02
\question Find all possible triangles 
(give the sides to 3 decimal places and the angles to 1 decimal place) with
\begin{multicols}{3}
	\begin{parts}
		\part $a=3$, $b=5$, $A=32^{\circ}$
		\part $b=2$, $c=4$, $C=63^{\circ}$
		\part $c=2$, $a=1$, $B=108^{\circ}$
	\end{parts}
\end{multicols}
\begin{solution}
	\begin{parts}
		%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
		%% Part a %%
		%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
		\part $a=3$, $b=5$, $A=32^{\circ}$
		\[
			\frac{3}{\sin{32^{\circ}}}
			=
			\frac{5}{\sin{B}}
			=
			\frac{c}{\sin{C}}
		\]
		\[
			B = \sin^{-1} \left( \frac{5}{3} \sin{32^{\circ}} \right)
		\]
		\[
			B = 62.030670\ldots ^{\circ}
		\]
		\castDiagram{62.0}{118.0}{}{}
		\[
			\underline{
				B = 62.030670\ldots ^{\circ}
				\quad \text{or} \quad
				B = 117.9693\ldots ^{\circ}
			}
		\]
		\[
			C = 180 - (A+B)
		\]
		\begin{multicols}{2}
			\noindent
			\begin{align*}
				\rightarrow C 
					&= 180 - (32+62.030670\ldots) 
					\\
				C 
					&= 85.96933\ldots ^{\circ}
			\end{align*}
			\begin{align*}
				\rightarrow C 
					&= 180 - (32+117.9693\ldots) 
					\\
				C 
					&= 30.0307\ldots ^{\circ}
			\end{align*}
		\end{multicols}
		\vspace{-1\baselineskip}
		\[
			\underline{
				C = 85.96933\ldots ^{\circ}
				\quad \text{or} \quad
				C = 30.0307\ldots ^{\circ}
			}
		\]
		We know that
		\[
			c 
			= 
			\left( \frac{3}{\sin{32^{\circ}}} \right) 
			\cdot
			\left( \sin{C} \right)
		\]
		\begin{multicols}{2}
			\noindent
			% \textul{Case 1:}
			\begin{align*}
				\rightarrow c 
					&= 
					\left( \frac{3}{\sin{32^{\circ}}} \right) 
					\cdot
					\sin{(85.96933\ldots^{\circ})}
					\\
				c 
					&= 5.64723\ldots
			\end{align*}
			% \textul{Case 2:}
			\begin{align*}
				\rightarrow c 
					&= 
					\left( \frac{3}{\sin{32^{\circ}}} \right) 
					\cdot
					\sin{(30.0307\ldots^{\circ})}
					\\
				c 
					&= 2.83324\ldots
			\end{align*}
		\end{multicols}
		\qSol{There are two possible triangles.}{}
		\qSolMath{
			1: 
			B = 62.0 \; (\text{1dp}), 
			\quad 
			C = 86.0 \; (\text{1dp}), 
			\quad 
			c = 5.647 \;(\text{3dp}) 
		}{}
		\qSolMath{
			2: 
			B = 118.0 \; (\text{1dp}), 
			\quad 
			C = 30.0 \; (\text{1dp}), 
			\quad 
			c = 2.833 \;(\text{3dp}) 
		}{}

		%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
		%% Part b %%
		%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
		\part $b=2$, $c=4$, $C=63^{\circ}$
		\[
			\frac{\sin{64^{\circ}}}{4}
			=
			\frac{\sin{B}}{2}
			=
			\frac{\sin{A}}{a}
		\]
		\begin{align*}
			\sin{B} 
				&= b \cdot \frac{\sin{C}}{c}
				\\
				&= \frac{1}{2} \sin{63^{\circ}}
				\\
			B 
				&= 26.455541\ldots ^{\circ}
		\end{align*}
		\castDiagram{26.5}{153.5}{}{}
		\[
			\underline{
				B = 26.455541\ldots ^{\circ}
				\quad \text{or} \quad
				B = \cancel{153.54445\ldots ^{\circ}}
			}
		\]
		But $153.5 + 63 > 180$
		\newline
		\underline{$\therefore$ only one possible triangle}
		\begin{align*}
			A 
				&= 180 - (B + C)
				\\
			\rightarrow A 
				&= 180 - (63+26.455541\ldots) 
				\\
			A 
				&= 90.5444\ldots ^{\circ}
		\end{align*}
		\vspace{-1.5\baselineskip}
		We know that
		\begin{align*}
			a 
				&= 
				\left( \frac{c}{\sin{C}} \right) 
				\cdot
				\left( \sin{C} \right)
				\\
			\rightarrow a 
				&= 
				\left( \frac{4}{\sin{63^{\circ}}} \right) 
				\cdot
				\sin{(90.5444\ldots)}
				\\
			a 
				&= 4.489102\ldots
		\end{align*}
		\qSolMath{
			A = 90.5^{\circ} \; (\text{1dp}), 
			\quad 
			B = 26.5^{\circ} \; (\text{1dp}), 
			\quad 
			a = 4.489 \;(\text{3dp}) 
		}{}

		\vspace{\baselineskip}
		\hl{BE note:} common student query ($>3$ times via email)
		was about rounded values in the calculation (probably due to rounded 
		values in the OV1115 booklet [messy for this topic]).
		If one uses the following \textit{rounded} values:
		\[
			B = 27^{\circ}, \quad A = 90^{\circ}
		\]
		\[
			a 
			= \left( \frac{4}{\sin{63^{\circ}}} \right) \cdot \sin{(90)}
		\]
		\[
			a = 4.489\textbf{304\ldots}
		\]

		%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
		%% Part c %%
		%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
		\part $c=2$, $a=1$, $B=108^{\circ}$
		\begin{align*}
			b^{2}
				&= c^{2} + a^{2} - 2ac \cos{B}
				\\
			b^{2}
				&= (2)^{2} + (1)^{2} 
				- \left( 2(1)(2) \cos{108^{\circ}} \right)
				\\
			b^{2}
				&= 6.2360679\ldots
				\\
			b
				&= 2.497212\ldots	
		\end{align*}
		\vspace{-1.5\baselineskip}
		\begin{align*}
			\cos{A} 
				&= \frac{b^{2} + c^{2} - a^{2}}{2bc} 
				\\
			\rightarrow \cos{A} 
				&= \frac{(6.2360679\ldots) + (2)^{2} - (1)^{2}}
				{2(2.497212\ldots)(2)} 
				\\
			% 0.92463795
			\rightarrow A 
				&= \cos^{-1}{\left( 0.924637\ldots \right)}
				\\
			A 
				&= 22.38617\ldots ^{\circ}
				\\
			C 
				&= 180 - (108 + 22.38617\ldots)
				\\
			C 
				&= 49.61382\ldots ^{\circ}
		\end{align*}
		\qSolMath{
			A = 22.4^{\circ} \; (\text{1dp}), 
			\quad 
			B = 49.6^{\circ} \; (\text{1dp}), 
			\quad 
			b = 2.497 \;(\text{3dp}) 
		}{}

	\end{parts}
\end{solution}
