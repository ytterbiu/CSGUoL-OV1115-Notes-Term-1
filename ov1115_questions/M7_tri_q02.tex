%% Triangle Formulae %%
% Exercise 1 - Q02
\question Find the angles (to 1 decimal place) in the following triangles
\label{q:triangle_formulae_q02}
\begin{multicols}{3}
	\begin{parts}
		\part $a=2$, $b=3$, $c=4$
		\part $a=1$, $b=1$, $c=1.5$
		\part $a=2$, $b=2$, $c=3$
	\end{parts}
\end{multicols}
\begin{solution}
	\begin{parts}
		\part $a=2$, $b=3$, $c=4$
		\begin{multicols}{3}
			\noindent
			\begin{align*}
				\cos{A}
					&= \frac{b^{2}+c^{2}-a^{2}}{2bc}
					\\
				A
					&= \cos^{-1} \left( \tfrac{9+16-4}{2(3)(4)} \right)
					\\
					&= \cos^{-1} \left( 0.875 \right)
					\\
					&= 28.955024\ldots
			\end{align*}
			\begin{align*}
				\cos{B}
					&= \frac{a^{2}+c^{2}-b^{2}}{2ac}
					\\
				B
					&= \cos^{-1} \left( \tfrac{4+16-9}{2(2)(4)} \right)
					\\
					&= \cos^{-1} \left( 0.6875 \right)
					\\
					&= 46.567463\ldots
			\end{align*}
			\begin{align*}
				\cos{C}
					&= \frac{a^{2}+b^{2}-c^{2}}{2ab}
					\\
				C
					&= \cos^{-1} \left( \tfrac{4+9-16}{2(2)(3)} \right)
					\\
					&= \cos^{-1} \left( -0.25 \right)
					\\
					&= 104.47751\ldots
			\end{align*}
		\end{multicols}
		\qSolMath{
			A = 29.0^{\circ},
			\quad
			B = 46.6^{\circ},
			\quad
			C = 104.5^{\circ}
		}{
			(1dp)
		}

		\part $a=1$, $b=1$, $c=1.5$
		\begin{multicols}{3}
			\noindent
			\begin{align*}
				\cos{A}
					&= \frac{b^{2}+c^{2}-a^{2}}{2bc}
					\\
				A
					&= \cos^{-1} \left( \tfrac{1+2.25-1}{2(1)(1.5)} \right)
					\\
					&= \cos^{-1} \left( 0.75 \right)
					\\
					&= 41.409622\ldots
			\end{align*}
			\begin{align*}
				\cos{B}
					&= \frac{a^{2}+c^{2}-b^{2}}{2ac}
					\\
				B
					&= \cos^{-1} \left( \tfrac{1+2.25-1}{2(1)(1.5)} \right)
					\\
					&= \cos^{-1} \left( 0.75 \right)
					\\
					&= 41.409622\ldots
			\end{align*}
			\begin{align*}
				\cos{C}
					&= \frac{a^{2}+b^{2}-c^{2}}{2ab}
					\\
				C
					&= \cos^{-1} \left( \tfrac{1+1-2.25}{2(1)(1)} \right)
					\\
					&= \cos^{-1} \left( -0.125 \right)
					\\
					&= 97.180755\ldots
			\end{align*}
		\end{multicols}
		\qSolMath{
			A = 41.4^{\circ},
			\quad
			B = 41.4^{\circ},
			\quad
			C = 97.2^{\circ}
		}{
			(1dp)
		}

		\part $a=2$, $b=2$, $c=3$
		\begin{multicols}{3}
			\noindent
			\begin{align*}
				\cos{A}
					&= \frac{b^{2}+c^{2}-a^{2}}{2bc}
					\\
				A
					&= \cos^{-1} \left( \tfrac{4+9-4}{2(2)(3)} \right)
					\\
					&= \cos^{-1} \left( 0.75 \right)
					\\
					&= 41.409622\ldots
			\end{align*}
			\begin{align*}
				\cos{B}
					&= \frac{a^{2}+c^{2}-b^{2}}{2ac}
					\\
				B
					&= \cos^{-1} \left( \tfrac{4+9-4}{2(2)(3)} \right)
					\\
					&= \cos^{-1} \left( 0.75 \right)
					\\
					&= 41.409622\ldots
			\end{align*}
			\begin{align*}
				\cos{C}
					&= \frac{a^{2}+b^{2}-c^{2}}{2ab}
					\\
				C
					&= \cos^{-1} \left( \tfrac{4+4-9}{2(2)(2)} \right)
					\\
					&= \cos^{-1} \left( -0.125 \right)
					\\
					&= 97.180755\ldots
			\end{align*}
		\end{multicols}
		\qSolMath{
			A = 41.4^{\circ},
			\quad
			B = 41.4^{\circ},
			\quad
			C = 97.2^{\circ}
		}{
			(1dp)
		}
	\end{parts}
\end{solution}
